query:--query
SET QUOTED_IDENTIFIER OFF
declare @sql nvarchar(MAX) = "[1]"

set @sql=SUBSTRING(@sql,2,LEN(@sql)-2)
set @sql=replace(@sql,"~#^","'")
set @sql=replace(@sql,"*~*","+")
set @sql=replace(@sql,"@#~",",")

if(len(@sql)>0)
begin
	BEGIN TRY
			EXEC(@sql)		
	END TRY
	BEGIN CATCH
		select 'failure' terr
	END CATCH
end
;
