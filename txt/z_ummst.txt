begin
	if(@t_custno != @custno)
	begin
		set @nowOrder = 1
		set @thisPageno = 1
		set @t_custno = @custno
	end
	else
	begin
		if(@nowOrder = (@pageCount))

			set @nowOrder = 1

		end
		else
		begin
			set @nowOrder = @nowOrder+1
		end
	end
	update #result set orderno=@nowOrder,pageno=@thisPageno where idno=@idno
	fetch next from cursor_table
	into @idno,@gno,@custno
end

	begin
